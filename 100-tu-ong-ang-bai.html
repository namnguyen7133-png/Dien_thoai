async def process_video_pipeline(video_path):
    """
    Xử lý 1 video từ A-Z: Transcription → AI Analysis → Upload
    """
    try:
        # 1. Transcription
        transcript = await transcribe_video(video_path)
        
        # 2. AI Analysis (Song song)
        tasks = [
            analyze_with_gemini(transcript),
            rewrite_with_chatgpt(transcript),
            optimize_with_grok(transcript)
        ]
        results = await asyncio.gather(*tasks)
        
        # 3. Tạo thumbnail
        thumbnail = await generate_thumbnail(results[0]['summary'])
        
        # 4. Đăng lên Facebook
        post_url = await upload_to_facebook(
            video_path, 
            results[1]['caption'],
            thumbnail
        )
        
        return {"success": True, "post_url": post_url}
        
    except Exception as e:
        print(f"❌ Lỗi: {e}")
        return {"success": False, "error": str(e)}
